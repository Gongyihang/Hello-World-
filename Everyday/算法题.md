//快速排序
class Solution3{
    public:
    void quick_sort(vector<int>& arr, int l, int h){
        if(l >= h) return ;
        int i = l - 1, j = h + 1, x = arr[(l + h) >> 1];
        while(i < j){
            do i++; while(arr[i] < x);
            do j--; while(arr[j] > x);
            if(i < j){
                swap(arr[i], arr[j]);
            }
        }
        quick_sort(arr, l, j);
        quick_sort(arr, j + 1, h);
    }
};


int main(){
    vector<int> arr = {40, 8, 7, 20, 5, 6, 7, 8, 9, 1};
    vector<int> res;
    Solution3 s3;
    s3.quick_sort(arr, 0, arr.size() - 1);
    for(int i = 0;i < 4;i++){
        cout << arr[i] << endl;
    }
    system("pause");
    return 0;
}

//括号匹配  
class Solution {
public:
    bool isValid (string const& s) {
        // 定义左右两边的括号序列
        string left = "([{";
        string right = ")]}";
        stack<char> stk;
        for (auto c : s) {
            // 判断每一个输入的字符是否为左括号，如果是就压栈
            if (left.find(c) != string::npos) {
                stk.push (c);
            } else {
                // 如果不是左括号，且如果发现栈为空，或者栈顶元素不是匹配的左括号的话，就返回 false
                if (stk.empty () || stk.top () != left[right.find (c)])
                    return false;
                // 如果匹配的话，就把栈顶出栈
                else
                    stk.pop ();
            }
        }
        return stk.empty();
    }
};

class Solution {
public:
    int strToInt(string str) {
        int res = 0, flag = 1, i = 0;
        while(str[i] == ' ') i++;
        if(str[i] == '-') flag = -1;
        if(str[i] == '-' || str[i] == '+') i++;
        for(; i < (int)str.size() && isdigit(str[i]); i++){
            if(res > INT_MAX / 10 || (res == INT_MAX / 10 && str[i] - '0' > 7)) 
                return flag == 1 ? INT_MAX : INT_MIN;
            res = res * 10 + (str[i] - '0');
        }
        return flag * res;
    }
};

bool ispair(char a, char b)
{
    if (a == '(' && b == ')')
        return true;
    if (a == '{' && b == '}')
        return true;
    if (a == '[' && b == ']')
        return true;
    return false;
}

int main()
{
    string s = "";
    while (getline(cin,s)) {
        if (s.empty()) {
            cout << "true" << endl;
            continue;
        } else {
            stack<int> st;
            st.push(s[0]);
            for (int i = 1; i < (int)s.size(); i++) {
                if (!st.empty() && ispair(st.top(), s[i]))
                    st.pop();
                else
                    st.push(s[i]);
            }
            if (st.empty())
                cout << "true" << endl;
            else
                cout << "false" << endl;
        }
    }
    return 0;
}
